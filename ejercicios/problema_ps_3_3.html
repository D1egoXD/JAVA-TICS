package com.tutortics;

import java.util.Scanner;

/**
 * Problema PS 3.3 - Contar Pares e Impares
 * Dado N n√∫meros enteros, determina cu√°ntos son pares y cu√°ntos impares.
 * Programaci√≥n Estructurada: todos los m√©todos son est√°ticos.
 */
public class problema_ps_3_3 {

    /**
     * Clase para almacenar el resultado del conteo de pares e impares.
     */
    public static class ResultadoConteo {
        public int pares;
        public int impares;
        
        public ResultadoConteo(int pares, int impares) {
            this.pares = pares;
            this.impares = impares;
        }
    }

    /**
     * Cuenta cu√°ntos n√∫meros en el arreglo son pares y cu√°ntos son impares.
     * 
     * @param numeros Arreglo de n√∫meros enteros
     * @param n Cantidad de n√∫meros a procesar
     * @return Objeto ResultadoConteo con pares e impares
     */
    public static ResultadoConteo contarParesImpares(int[] numeros, int n) {
        int pares = 0;     // PARES ‚Üê 0
        int impares = 0;   // IMPARES ‚Üê 0
        
        // I ‚Üê 1, mientras I ‚â§ N
        for (int i = 0; i < n; i++) {
            int num = numeros[i];  // Leer NUM
            
            if (num % 2 == 0) {  // Si NUM % 2 = 0
                pares = pares + 1;  // PARES ‚Üê PARES + 1
            } else {
                impares = impares + 1;  // IMPARES ‚Üê IMPARES + 1
            }
            // I ‚Üê I + 1 (autom√°tico en el for)
        }
        
        return new ResultadoConteo(pares, impares);
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("========================================");
        System.out.println("Contador de N√∫meros Pares e Impares");
        System.out.println("========================================");
        
        // Lectura de N
        System.out.print("¬øCu√°ntos n√∫meros desea ingresar? N = ");
        int n = scanner.nextInt();

        // Leer los N n√∫meros
        int[] numeros = new int[n];
        System.out.println("\nIngrese los " + n + " n√∫meros:");
        for (int i = 0; i < n; i++) {
            System.out.print("N√∫mero " + (i + 1) + ": ");
            numeros[i] = scanner.nextInt();
        }

        // Contar pares e impares
        ResultadoConteo resultado = contarParesImpares(numeros, n);

        // Mostrar resultados
        System.out.println("\n========================================");
        System.out.println("Resultados:");
        System.out.println("N√∫meros PARES: " + resultado.pares);
        System.out.println("N√∫meros IMPARES: " + resultado.impares);
        System.out.println("========================================");

        scanner.close();
    }
}
