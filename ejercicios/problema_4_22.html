package com.tutortics;

import java.util.Scanner;

public class problema_4_22 {

    // M√©todo para encontrar el producto con mayor existencia
    public static int obtenerMayorExistencia(int[] can, int n) {
        int max = can[0];
        int pos = 0;
        
        for (int i = 1; i < n; i++) {
            if (can[i] > max) {
                max = can[i];
                pos = i;
            }
        }
        
        return pos;
    }

    // M√©todo para imprimir productos a pedir
    public static void productosAPedir(int[] pro, int[] can, int n) {
        System.out.println("\n=== PRODUCTOS A PEDIR AL PROVEEDOR ===");
        boolean hayProductos = false;
        
        for (int i = 0; i < n; i++) {
            if (can[i] >= 1 && can[i] < 50) {
                int cantidad = 50 - can[i];
                System.out.println("Clave: " + pro[i] + " - Pedir: " + cantidad + " cajas");
                hayProductos = true;
            }
        }
        
        if (!hayProductos) {
            System.out.println("No hay productos que pedir");
        }
    }

    // M√©todo para eliminar productos con existencia cero
    public static int eliminarConExistenciaCero(int[] pro, int[] can, int n) {
        int j = 0;
        
        for (int i = 0; i < n; i++) {
            if (can[i] != 0) {
                pro[j] = pro[i];
                can[j] = can[i];
                j++;
            }
        }
        
        return j; // Nuevo tama√±o
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        // Lectura del n√∫mero de productos
        System.out.print("Ingrese el n√∫mero de productos (1-100): ");
        int n = scanner.nextInt();

        if (n < 1 || n > 100) {
            System.out.println("Error: N debe estar entre 1 y 100");
            scanner.close();
            return;
        }

        int[] pro = new int[n];
        int[] can = new int[n];

        // a) Lectura de claves y cantidades
        System.out.println("\n=== INGRESO DE PRODUCTOS ===");
        for (int i = 0; i < n; i++) {
            System.out.print("Producto " + (i + 1) + " - Clave: ");
            pro[i] = scanner.nextInt();
            System.out.print("Producto " + (i + 1) + " - Existencia: ");
            can[i] = scanner.nextInt();
        }

        // b) Producto con mayor existencia
        int posMayor = obtenerMayorExistencia(can, n);
        System.out.println("\n=== PRODUCTO CON MAYOR EXISTENCIA ===");
        System.out.println("Clave: " + pro[posMayor] + " - Existencia: " + can[posMayor]);

        // c) Productos a pedir al proveedor
        productosAPedir(pro, can, n);

        // d) Eliminar productos con existencia cero
        int nuevoN = eliminarConExistenciaCero(pro, can, n);
        
        System.out.println("\n=== INVENTARIO DESPU√âS DE ELIMINAR EXISTENCIA CERO ===");
        System.out.println("Productos restantes: " + nuevoN);
        for (int i = 0; i < nuevoN; i++) {
            System.out.println("Clave: " + pro[i] + " - Existencia: " + can[i]);
        }

        scanner.close();
    }
}
